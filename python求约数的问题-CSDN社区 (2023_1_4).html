def showMaxFactor(num):
count=num/2
while count>1:
if num%count==0:
print'largest factor of %d is %d'% (num,count)
break
count-=1
else:
print num, "is prime"
#调用这个函数：
for i in range(10,30):
showMaxFactor(i)
